    function initPriceAccordionAnimation() {
      const details = document.getElementById('price-explanation');
      if (!details || details.dataset.animationReady === 'true') return;
      const summary = details.querySelector('summary');
      const content = details.querySelector('.digi-price-breakdown-pakkeguide');
      if (!summary || !content) return;
      details.dataset.animationReady = 'true';

      summary.addEventListener('click', function(event) {
        event.preventDefault();
        if (details.dataset.animating === 'true') return;

        if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
          details.open = !details.open;
          return;
        }

        const isClosing = details.open;
        const startHeight = details.offsetHeight;
        if (!isClosing) details.open = true;
        const endHeight = isClosing ? summary.offsetHeight : summary.offsetHeight + content.offsetHeight;

        details.dataset.animating = 'true';
        details.style.overflow = 'hidden';
        const animation = details.animate(
          { height: [`${startHeight}px`, `${endHeight}px`] },
          { duration: 250, easing: 'cubic-bezier(0.4, 0, 0.2, 1)' }
        );

        animation.addEventListener('finish', function() {
          if (isClosing) details.open = false;
          details.style.height = '';
          details.style.overflow = '';
          details.dataset.animating = 'false';
        }, { once: true });

        animation.addEventListener('cancel', function() {
          details.style.height = '';
          details.style.overflow = '';
          details.dataset.animating = 'false';
        }, { once: true });
      });
    }

        } else {
          step6Slide.classList.add('digi-step-hidden-pakkeguide');
        }
      }
      
      // Get visible slides in order
      const visibleSlides = getVisibleStepOrder();
      const targetIndex = visibleSlides.indexOf(stepNumber);
      
      if (targetIndex !== -1) {
        // Apply transform based on visible slides
        container.style.transform = `translateX(-${targetIndex * 100}%)`;
      }
      
      currentStep = stepNumber;
      const visibleSteps = getVisibleStepOrder();
      const currentIndex = visibleSteps.indexOf(stepNumber);
      maxVisitedStep = Math.max(maxVisitedStep, currentIndex + 1);
      updateProgressBar();

      }

    function getVisibleStepOrder() {
      const shouldShow4 = SHOW_SERVICEAFTALE_STEP && answers.q3 === 'Eje';  // Step 4 (warranty) only shown if ownership is "Eje" AND feature enabled
      const shouldShow6 = USED_CHARGER_ENABLED;  // Step 6 (used charger) based on feature flag
      
      let steps = [1, 3];
      if (shouldShow4) steps.push(4);
      steps.push(2);
      steps.push(5);
      
      if (shouldShow6) {
        steps.push(6);
      }
      
      steps.push('result'); // Always add result step at the end
      
      return steps;
    }

    function getNextStep() {
      const visibleSteps = getVisibleStepOrder();
      const currentIndex = visibleSteps.indexOf(currentStep);
      if (currentIndex < visibleSteps.length - 1) {
        return visibleSteps[currentIndex + 1];
      }
      return null;
    }

    function updateProgressBar() {
      const visibleSteps = getVisibleStepOrder();
      const currentIndex = visibleSteps.indexOf(currentStep);
      const totalSteps = visibleSteps.length;
      const progress = ((currentIndex + 1) / totalSteps) * 100;
      document.getElementById('progress-fill').style.width = `${progress}%`;
    }

        if (note) note.style.display = 'none';

        step6Slide.querySelectorAll('input').forEach(i => { i.disabled = true; i.checked = false; });
        if (note) note.style.display = 'none';
        answers.q6 = 'Nej';

    function handleAnswer(step) {
      const selected = document.querySelector(`input[name="q${step}"]:checked`);
      if (!selected) return;

      answers[`q${step}`] = selected.value;
      if (step === 1 || step === 'cable') updateOwnershipStartupPrices();
      const continueButton = document.querySelector('.digi-step-continue-pakkeguide[data-step="' + step + '"]');
      if (continueButton) continueButton.disabled = false;

      // Handle special logic for each step
      if (step === 3) {
        // Handle question 4 logic based on Eje/Leje choice
        if (selected.value === 'Leje') {
          delete answers.q4;
          answers.q4 = 'Nej'; // Default for leje - skip step 4
        } else if (selected.value === 'Eje' && !SHOW_SERVICEAFTALE_STEP) {
          // If serviceaftale step is hidden, don't set a default answer for q4
          delete answers.q4;
        }
        // If "Eje" is selected, we'll navigate to step 4 next (if enabled)
      } else if (step === 5 && USED_CHARGER_ENABLED) {
        // Handle Q6 (used charger) logic
        if (answers.q5 === 'Nej') {
          answers.q6 = 'Nej';
          qsa('.digi-q4note-pakkeguide').forEach(n => n.style.display = 'block');
        } else {
          qsa('.digi-q4note-pakkeguide').forEach(n => n.style.display = 'none');
        }
      }

      if (step === 3 || step === 'cable') updateOwnershipContinueState();

    }

    function updateOwnershipContinueState() {

    }

      if (nextStep) showStep(nextStep);
    }

    function calculateResultNew() {
      // New configurable pricing logic
      const isRent = answers.q3 === 'Leje';
      const isOwn = answers.q3 === 'Eje';
      const hasServiceAgreement = isRent || (isOwn && answers.q4 === 'Ja');
      const hasCable = answers.q1 === 'Peblar';
      const cableLength = answers.qcable || '5';
      const hasElectricity = answers.q5 === 'Ja';
      const isExistingInstallation = answers.q2 === 'Ja';
      const effectiveQ6 = USED_CHARGER_ENABLED ? answers.q6 : 'Nej';
      
      let startup = 0;
      let monthly = 0;
      
      // Base pricing
      if (isRent) {
        startup += PRICING_CONFIG.RENT_BASE.startup;
        monthly += PRICING_CONFIG.RENT_BASE.monthly;
      } else if (isOwn) {
        startup += PRICING_CONFIG.OWN_BASE.startup;
        monthly += PRICING_CONFIG.OWN_BASE.monthly;
        
        // Add service agreement cost if selected for ownership
        if (hasServiceAgreement) {
          monthly += PRICING_CONFIG.SERVICE_AGREEMENT_MONTHLY;
        }
      }
      
      // Add-ons
      if (hasCable) {
        startup += PRICING_CONFIG.CABLE_UPGRADE_STARTUP;
        if (cableLength === '7') startup += PRICING_CONFIG.CABLE_7M_UPGRADE_STARTUP;
      }
      
      if (hasElectricity) {
        monthly += PRICING_CONFIG.ELECTRICITY_AGREEMENT_MONTHLY;
      } else {
        startup += PRICING_CONFIG.NO_ELECTRICITY_STARTUP;
        if (!isOwn) monthly += PRICING_CONFIG.NO_ELECTRICITY_MONTHLY;
      }
      
      // Rabat ved eksisterende installation (kun eje)
      if (isExistingInstallation && isOwn) {
        startup = Math.max(0, startup - PRICING_CONFIG.EXISTING_INSTALLATION_STARTUP_DISCOUNT);
      }
      
      // Used charger discount
      if (effectiveQ6 === 'Ja') {
        startup = Math.max(0, startup - PRICING_CONFIG.USED_CHARGER_DISCOUNT);
      }
      
      // Generate link with subscription parameter
      const link = generateOrderLink();
      
      // Update UI
      document.getElementById('digi-startup-price').textContent = startup.toLocaleString('da-DK');
      document.getElementById('digi-monthly-price').textContent = monthly.toLocaleString('da-DK');
      document.getElementById('digi-cta-link').href = link;
      renderResultProduct();
      renderPriceBreakdown({ monthly, isRent, isOwn, hasServiceAgreement, hasElectricity });
      
      // Add "Fra " prefix if Eje and serviceaftale step is hidden (user hasn't decided yet)
      const showFromPrefix = isOwn && !SHOW_SERVICEAFTALE_STEP;
      document.getElementById('price-prefix-startup').textContent = showFromPrefix ? 'Fra ' : '';
      document.getElementById('price-prefix-monthly').textContent = showFromPrefix ? 'Fra ' : '';
      
    }

    function renderPriceBreakdown({ monthly, isRent, isOwn, hasServiceAgreement, hasElectricity }) {
      const baseMonthly = isRent ? PRICING_CONFIG.RENT_BASE.monthly : PRICING_CONFIG.OWN_BASE.monthly;
      const serviceMonthly = isOwn && hasServiceAgreement ? PRICING_CONFIG.SERVICE_AGREEMENT_MONTHLY : 0;
      const electricityMonthly = hasElectricity
        ? PRICING_CONFIG.ELECTRICITY_AGREEMENT_MONTHLY
        : (isOwn ? 0 : PRICING_CONFIG.NO_ELECTRICITY_MONTHLY);
      document.getElementById('digi-breakdown-base-label').textContent = isRent
        ? 'Leje inkl. serviceaftale'
        : 'Køb af ladeboks';
      document.getElementById('digi-breakdown-base-monthly').textContent = baseMonthly.toLocaleString('da-DK');
      document.getElementById('digi-breakdown-base-row').hidden = isOwn && hasElectricity;
      document.getElementById('digi-breakdown-service-monthly').textContent = serviceMonthly.toLocaleString('da-DK');
      document.getElementById('digi-breakdown-electricity-label').textContent = hasElectricity
        ? 'OK El Højt Forbrug abonnement'
        : 'Tillæg uden OK Elaftale';
      document.getElementById('digi-breakdown-electricity-monthly').textContent = electricityMonthly.toLocaleString('da-DK');
      document.getElementById('digi-breakdown-service-row').hidden = serviceMonthly === 0;
      document.getElementById('digi-breakdown-electricity-row').hidden = electricityMonthly === 0;
      document.getElementById('digi-breakdown-monthly-section').hidden = monthly === 0;
    }
    
    function renderResultProduct() {
      const isPeblar = answers.q1 === 'Peblar';
      const image = document.getElementById('digi-result-product-image');
      const name = document.getElementById('digi-result-product-name');
      const manufacturer = document.getElementById('digi-result-spec-manufacturer');
      const country = document.getElementById('digi-result-spec-country');
      const size = document.getElementById('digi-result-spec-size');
      const fixedCable = document.getElementById('digi-result-spec-fixed-cable');
      const cableLength = document.getElementById('digi-result-spec-cable-length');
      const cableLengthRow = document.getElementById('digi-result-spec-cable-length-row');
      const connectorLabel = document.getElementById('digi-result-spec-connector-label');
      if (!image || !name || !manufacturer || !country || !size || !fixedCable || !cableLength || !cableLengthRow || !connectorLabel) return;

      image.src = isPeblar
        ? 'https://www.ok.dk/imagevault/publishedmedia/kdl5jnnf5ze5xgdnhkpo/peblar-sort-v-g.png'
        : 'https://www.ok.dk/imagevault/publishedmedia/ge62mn9w5obn02ll5jxx/domo-mursten.png';
      image.alt = isPeblar ? 'Peblar-ladeboks' : 'Domo-ladeboks';
      name.textContent = isPeblar ? 'Peblar' : 'Domo';
      manufacturer.textContent = isPeblar ? 'Peblar' : 'Sargo';
      country.textContent = isPeblar ? 'EU' : 'Danmark';
      size.textContent = isPeblar ? '207 × 387 × 128 mm' : '190 × 338 × 116 mm';
      fixedCable.textContent = isPeblar ? 'Ja' : 'Nej';
      cableLength.textContent = isPeblar ? ((answers.qcable || '5') + ' m') : '—';
      cableLengthRow.hidden = !isPeblar;
      connectorLabel.textContent = isPeblar ? 'Ladestik' : 'Udtag';
    }

    function generateOrderLink() {
      // Generate order link with all parameters including subscription
      const baseUrl = 'https://www.ok.dk/privat/bestil/ladestander-bestillingsformular';
      const params = new URLSearchParams();
      
      // Installation type
      params.append('eksisterende-installation', answers.q2 === 'Ja' ? '1' : '2');
      
      // Ownership type
      params.append('ladeboks', answers.q3.toLowerCase());
      
      // Cable type
      params.append('kabel', answers.q1 === 'Peblar' ? '1' : '2');
      
      // Subscription parameter
      const hasServiceAgreement = answers.q3 === 'Leje' || (answers.q3 === 'Eje' && answers.q4 === 'Ja');
      params.append('serviceaftale', hasServiceAgreement ? '1' : '0');
      
      // Electricity agreement
      if (answers.q5 === 'Ja') {
        params.append('elaftale', '6354');
        params.append('elaftale', '6356');
      }
      
      // Source parameter (always last)
      params.append('source', 'ladepakkeberegner');
      
      return `${baseUrl}?${params.toString()}`;
    }
